export default async function handler(req, res) { if (req.method !== "POST") { return res.status(405).json({ status: "error", message: "Method not allowed" }); } try { const { number, message, instance_id, access_token } = req.body || {}; if (!number || !message || !instance_id || !access_token) { return res.status(400).json({ status: "error", message: "Missing required parameters from page", received: { number, message, instance_id, access_token } }); } const tries = []; // 1) JSON try { const r1 = await fetch("https://whatsbot.at/whatsapp_livechat/send_message", { method: "POST", headers: { "Content-Type": "application/json", "Accept": "application/json" }, body: JSON.stringify({ number, message, instance_id, access_token }) }); const t1 = await r1.text(); tries.push({ mode: "json", status: r1.status, body: t1 }); if (r1.ok) { return res.status(200).send(t1); } } catch (e) { tries.push({ mode: "json", error: e.message }); } // 2) form-urlencoded try { const body2 = new URLSearchParams(); body2.append("number", number); body2.append("message", message); body2.append("instance_id", instance_id); body2.append("access_token", access_token); const r2 = await fetch("https://whatsbot.at/whatsapp_livechat/send_message", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded;charset=UTF-8", "Accept": "application/json,text/plain,*/*" }, body: body2.toString() }); const t2 = await r2.text(); tries.push({ mode: "urlencoded", status: r2.status, body: t2 }); if (r2.ok) { return res.status(200).send(t2); } } catch (e) { tries.push({ mode: "urlencoded", error: e.message }); } // 3) multipart/form-data try { const form = new FormData(); form.append("number", number); form.append("message", message); form.append("instance_id", instance_id); form.append("access_token", access_token); const r3 = await fetch("https://whatsbot.at/whatsapp_livechat/send_message", { method: "POST", body: form }); const t3 = await r3.text(); tries.push({ mode: "formdata", status: r3.status, body: t3 }); if (r3.ok) { return res.status(200).send(t3); } } catch (e) { tries.push({ mode: "formdata", error: e.message }); } return res.status(500).json({ status: "error", message: "All send attempts failed", tries }); } catch (error) { return res.status(500).json({ status: "error", message: error.message }); } }
